function openChartModal() { const modal = document.getElementById("chartModal"); modal.style.display = "flex"; // --- 1. Fetch real summary data for the Doughnut Chart --- fetch('/api/finance/summary') .then(response => response.json()) .then(data => { if (myChart !== null) myChart.destroy(); const ctx1 = document.getElementById('financeChart').getContext('2d'); myChart = new Chart(ctx1, { type: 'doughnut', data: { labels: ['Income', 'Expenses'], datasets: [{ data: [data.totalIncome, data.totalExpenses], backgroundColor: ['#2ecc71', '#e74c3c'], borderWidth: 1 }] } }); }) .catch(error => console.error("Error loading summary:", error)); // --- 2. Fetch real category data for the Bar Chart 🚀 --- fetch('/api/finance/analytics/category') .then(response => response.json()) .then(data => { // Extracts categories and totals dynamically from your backend! const labels = data.map(item => item.CATEGORY || item.category); const amounts = data.map(item => item.TOTAL || item.total); if (categoryChart !== null) categoryChart.destroy(); const ctx2 = document.getElementById('categoryChart').getContext('2d'); categoryChart = new Chart(ctx2, { type: 'bar', data: { labels: labels, // Real categories from DB! datasets: [{ label: 'Expenses ($)', data: amounts, // Real amounts from DB! backgroundColor: '#3498db', borderWidth: 1 }] }, options: { scales: { y: { beginAtZero: true } } } }); }) .catch(error => console.error("Error loading categories:", error)); // --- 3. Line Chart (Still using dummy data for now) --- if (trendChart !== null) trendChart.destroy(); const ctx3 = document.getElementById('trendChart').getContext('2d'); trendChart = new Chart(ctx3, { type: 'line', data: { labels: ['Nov', 'Dec', 'Jan', 'Feb', 'Mar', 'Apr'], datasets: [ { label: 'Income', data: [4000, 4000, 4500, 4200, 4500, 4800], borderColor: '#2ecc71', fill: false, tension: 0.1 }, { label: 'Expenses', data: [3100, 3500, 3000, 3200, 2800, 2900], borderColor: '#e74c3c', fill: false, tension: 0.1 } ] }, options: { maintainAspectRatio: false, scales: { y: { beginAtZero: true } } } }); }
